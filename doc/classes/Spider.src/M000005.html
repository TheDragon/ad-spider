  def find_omniture_tag(link)
    tags = Array.new
    omnisearch = @agent.get(link)
    omnisearch.search("//script[@language='JavaScript']").each do |block|
      if block.inner_html =~ /(s\.)/
        tags << block.inner_html.split("\n").grep(/^s\./)
      end
    end
    tags.reject! {|t| t.empty?}
    return tags.first
  end
